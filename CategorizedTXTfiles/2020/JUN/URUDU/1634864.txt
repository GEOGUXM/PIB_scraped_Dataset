      وزیراعظم کا دفتر

      وزیر اعظم کا معزز ڈاکٹر جوزیف مار تھوما میٹرو پولیٹن کی 90 ویں یومِ پیدائش کی تقریبات سے خطاب
      Posted On: 27 JUN 2020 11:58AM by PIB Delhi


    نئی دلّی ، 27 جون / عزت مآب ڈاکٹر جوزیف مار تھوما میٹرو پولیٹن ، معزز فادرس ، مارتھوما چرچ کے معزز ارکان ،


    یہ میرے لئے ایک بڑے اعزاز کی بات ہے کہ میں اس معزز اجتماع سے خطاب کر رہا ہوں ۔ ہم یہاں انتہائی معزز ڈاکٹر جوزیف مارتھوما میٹرو پولیٹن کے 90 ویں یومِ پیدائش کے مخصوص موقع پر جمع ہوئے ہیں ۔ میں انہیں اپنی طرف سے مبارکباد اور طویل عمر اور بہترین صحت کی نیک خواہشات پیش کرتا ہوں ۔ ڈاکٹر مارتھوما نے اپنی پوری زندگی ہمارے سماج اور ملک کی بہتری کے لئے وقف کر دی ہے ۔ وہ خاص طور پر غریبی کے خاتمے اور خواتین کو با اختیار بنانے کے لئے شدید جذبہ رکھتے ہیں ۔


    دوستو ،


    مار تھوما چرچ حضرت عیسیٰ مسیح کے شاگرد سینٹ تھومس کے عظیم نظریات سے جڑا ہوا ہے ۔ بھارت ہمیشہ ہی مختلف ذرائع سے روحانی اثرات قبول کرنے کے لئے ہمیشہ کھلا رہا ہے ۔ سینٹ تھومس کا تعاون بہت زیادہ ہے اور اُن کے نقشہ قدم پر چلتے ہوئے ہندوستانی عیسائی برادری قابلِ قدر ہے ۔ سینٹ تھومس کے ساتھ ہم عاجزی کو سمجھتے ہیں ۔ انہوں نے درست کہا ہے کہ عاجزی ایک نعمت ہے ، جو ہمیشہ اچھے کاموں کے لئے مفید ہے ۔ یہ وہ جذبہ ہے ، جس کے ساتھ مار تھوما چرچ نے ہمارے بھارتی ہم وطنوں کی زندگی میں مثبت تبدیلی لانے کے لئے کام کیا ہے ۔ انہوں نے صحت اور تعلیم جیسے شعبوں میں بہت کام کیا ہے ۔ سینٹ تھومس وسیع ذہانت کے مالک تھے ۔ مار تھوما چرچ نے بھارت کی جدو جہد آزادی میں اہم رول ادا کیا ۔ یہی چرچ قومی یکجہتی کو آگے بڑھانے کے لئے ہمیشہ آگے رہا ہے ۔


    اس چرچ نے ایمرجنسی میں لڑائی لڑی ۔ یہ ایک بڑے فخر کی بات ہے کہ مار تھوما چرچ کی جڑیں بھارتی اقدار میں پیوست ہیں ۔ چرچ کے تعاون کو قومی سطح پر بھی تسلیم کیا گیا ہے ۔ مار تھوما چرچ کے سابق میٹرو پولیٹن فلیپوس مار کریسوسٹوم کو 2018 ء میں پدم بھوشن سے نوازا گیا ۔ انہوں نے بہت سے لوگوں کو تحریک دی ۔


    دوستو ،


    دنیا عالمی وباء کے خلاف ایک مضبوط لڑائی لڑ رہی ہے ۔ کووڈ – 19 صرف ایک جسمانی بیماری نہیں ہے ، جس سے لوگوں کی زندگی کو خطرہ ہے بلکہ یہ ہمارے غیر صحت مند طرز زندگی کی جانب توجہ دلاتی ہے ۔ ایک عالمی وباء کا مطلب ہے کہ پوری انسانیت کو علاج کی ضرورت ہے ۔ ہمیں کرۂ ارض میں زیادہ ہم آہنگی اور خوشیوں کے لئے ہر ممکن کوشش کرنی چاہیئے ۔


    دوستو ،


    آپ کو یہ جان کر خوشی ہوگی کہ اپنے کورونا جانبازوں کی قوت کے ساتھ بھارت کووڈ – 19 سے سخت لڑائی لڑ رہا ہے ۔ اس سال کے شروع میں کچھ لوگوں نے پیش گوئی کی تھی کہ بھارت میں اِس وائرس کا اثر بہت زیادہ شدید ہو گا لیکن لاک ڈاؤن اور حکومت کے ذریعے کئے گئے بہت سے اقدامات کے ساتھ ساتھ عوام کی لڑائی کی وجہ سے بھارت آج کئی دیگر ملکوں کے مقابلے بہتر پوزیشن پر ہے ۔ بھارت کی صحت یابی کی شرح بڑھ رہی ہے ۔


    زندگی کا زیاں ، چاہے وہ کووڈ کی وجہ سے ہو یا کسی اور وجہ سے ، افسوسناک ہے ۔ البتہ بھارت میں فی 10 لاکھ اموات کی شرح 12 سے کم ہے ، جب کہ یہ اٹلی میں فی 10 لاکھ 574 ہے ۔ امریکہ ، برطانیہ ، اسپین اور فرانس میں بھی تعداد بھارت کے مقابلے کافی زیادہ ہے ۔ لاکھوں گاؤں ، جہاں تقریباً 85 کروڑ کی آبادی ہے ، کورونا وائرس سے تقریباً پوری طرح پاک ہیں ۔


    دوستو ،


    عوام کے ذریعے کووڈ – 19 کے خلاف لڑی گئی اِس لڑائی نے اب تک اچھے نتائج دیئے ہیں اور کیا ہمیں اپنی چوکسی کو کم کر دینا چاہیئے ، نہیں بالکل نہیں ، حقیقت میں ہمیں اور زیادہ محتاط رہنا چاہیئے ۔ ماسک پہننا چاہیئے ، دو گز کی دوری ، سوشل ڈسٹنسنگ ، بھیڑ بھاڑ والی جگہ پر جانے سے پرہیز اب اور زیادہ اہم ہو گیا ہے ۔


    اس کے ساتھ ہی ہمیں 130 کروڑ بھارتیوں کی اقتصادی ترقی اور خوش حالی پر توجہ دینی ہے ۔ تجارت و کامرس کے پہیے گھومنے چاہئیں ۔ زراعت میں فروغ ہونا چاہیئے ۔ پچھلے چند ہفتوں میں حکومتِ ہند نے معیشت سے متعلق طویل مدتی اور مختصر مدتی اقدامات کئے ہیں ۔ سمندر سے خلاء تک ، کھیت سے فیکٹریوں تک تمام شعبوں میں عوام دوست اور ترقی حامی فیصلے کئے گئے ہیں ۔


    آتم نربھر بھارت ، خود کفیل بھارت ہر ہندوستانی کے لئے اقتصادی قوت اور خوشحالی کو یقینی بنائے گا ۔ ایک مہینے پہلے مرکزی کابینہ نے پردھان منتری متسیہ سمپدا یوجنا کو منظوری دی ہے ۔ یہ اسکیم 20 ہزار کروڑ روپئے کی سرمایہ کاری کے ساتھ ماہی گیری کے سیکٹر میں تبدیلی لائے گی ، بر آمدات کی کمائی میں اضافہ کرے گی اور 55 لاکھ سے زیادہ لوگوں کو روز گار فراہم کرے گی ۔ ویلیو چین کو زیادہ مضبوط بنانے کے لئے بہتر ٹیکنا لوجی اور بنیادی ڈھانچہ فراہم کرایا جائے گا۔ مجھے یقین ہے کہ کیرالہ کے ماہی گیر اس اسکیم سے فائدہ اٹھائیں گے ۔


    دوستو ،


    خلاء کے سیکٹر میں تاریخی اصلاحات کی گئی ہیں ۔ یہ اصلاحات خلائی اثاثوں اور سرگرمیوں کے زیادہ استعمال کو یقینی بنائیں گی ۔ اعداد و شمار اور ٹیکنا لوجی تک رسائی بہتر ہو گی ۔ کیرالہ کے بہت سے جوان ، خاص طور پر جنوبی بھارت کے نو جوان ، جو سائنس اور ٹیکنا لوجی میں بہت دلچسپی رکھتے ہیں ، ان اصلاحات سے فائدہ اٹھایں گے ۔


    دوستو ،


    بھارت کو ترقی کا انجن بنانے کے لئے ہماری حکومت نے ہمیشہ احساس اور طویل مدتی ویژن سے رہنمائی حاصل کی ہے ۔ ہم نے یہ فیصلے دلّی میں سرکار کے آرام دہ دفاتر میں نہیں کئے بلکہ بنیادی طور پر لوگوں کی رائے کے بعد کئے ہیں ۔ اسی جذبے کے ساتھ ہر ہندوستانی کو ایک بینک کھاتے تک رسائی حاصل کرائی گئی ہے ، 8 کروڑ خاندانوں کو دھوئیں سے پاک باورچی خانے ملے ہیں ، 1.5 کروڑ بے گھر لوگوں کو مکان دیئے گئے ہیں ۔


    بھارت اب دنیا کی حفظانِ صحت کی سب سے بڑی اسکیم آیوشمان بھارت کا بھی مرکز ہے ۔ ایک کروڑ سے زیادہ لوگوں نے معیاری علاج کرایا ہے ۔ غریبوں کے لئے ہم نے ایک ملک اور راشن کارڈ اسکیم شروع کی ہے ، جس سے انہیں ، جہاں بھی وہ ہیں ، راشن حاصل کرنے میں مدد ملی ہے ۔ متوسط طبقے کے لئے رہن سہن میں آسانی کو فروغ دینے کے لئے کئی اقدامات کئے گئے ہیں ۔ کسانوں کے لئے ایم ایس پی میں اضافہ کیا گیا ہے اور اِس بات کو یقینی بنایا گیا ہے کہ انہیں درست قیمتیں ملیں اور یہ سیکٹر دلالوں سے پاک ہو ۔


    خواتین کے لئے اِس بات کو یقینی بنایا گیا ہے کہ اُن کی صحت پر مختلف اسکیموں کے ذریعے پوری توجہ دی جائے اور زچگی کی چھٹیوں میں اضافے کے ذریعے اُن کے کیریئر کو بھی محفوظ کیا گیا ہے ۔ بھارتی حکومت عقائد ، صنف ،ذات پات یا زبان کے درمیان امتیاز نہیں کرتی اور 130 کروڑ بھارتیوں کو با اختیار بنانے کی خواہش کے ساتھ آگے بڑھ رہی ہے اور ہماری رہنمائی کے لئے بھارت کا آئین ہے ۔


    دوستو ،


    مقدس بائبل وسیع طور پر اجتماعیت کے بارے میں بات کرتی ہے ۔ اب وقت آ گیا ہے کہ ہر طبقے کے لوگ ہمارے ملک کی ترقی کے لئے مل کر کام کریں ۔ اس بات کو سوچیں کہ کس طرح ہماری سرگرمیاں قومی ترقی میں تعاون کر سکتی ہیں ۔ آج بھارت کہہ رہا ہے کہ ہم" ہم مقامی طور پر پیدا کریں گے اور مقامی مصنوعات ہی خریدں گے " ۔ اس سے بہت سے گھروں میں خوشحالی کے دئے جلیں گے ۔


    ہمارے ملک کو مستحکم کرنے کے بہت سارے طریقے ہیں ۔ مار تھوما چرچ اپنی اقدار کے خطوط پر یقیناً اِس موقع پر آگے آئے گا اور آنے والے وقت میں بھارت کی ترقی میں اہم رول ادا کرے گا ۔


    ایک بار پھر میں محترم ڈاکٹر جوزیف مار تھوما میٹرو پولیٹن کو مبارکباد دیتا ہے ۔


    آپ سب کا شکریہ ، بہت بہت شکریہ ۔
